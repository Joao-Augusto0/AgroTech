GERADOR DE MOTORISTA

DELIMITER //

CREATE PROCEDURE insere_dados_motorista()
BEGIN
  DECLARE contador INT DEFAULT 0;
  WHILE contador < 100 DO
    SET @nome := CONCAT('Motorista ', contador);
    SET @cpf := FLOOR(RAND() * 99999999999);
    SET @cnh := FLOOR(RAND() * 999999999);
    INSERT INTO motorista (nome, cpf, cnh)
    SELECT @nome, @cpf, @cnh
    FROM DUAL
    WHERE NOT EXISTS (
      SELECT * FROM motorista
      WHERE cpf = @cpf OR cnh = @cnh
    )
    LIMIT 1;
    SET contador := contador + 1;
  END WHILE;
END //

DELIMITER ;

CALL insere_dados_motorista();

GERADOR DE FROTAS

-- Criar o procedimento armazenado
DELIMITER $$
CREATE PROCEDURE inserir_frota()
BEGIN
  DECLARE i INT DEFAULT 1;
  WHILE (i <= 100) DO
    INSERT INTO frota (modelo, placa, marca, tipo)
    VALUES (
      CONCAT('Modelo ', FLOOR(RAND() * 100)),
      CONCAT(UPPER(LEFT(CHAR(FLOOR(RAND() * 26) + 65), 3)), '-', FLOOR(RAND() * 10000)),
      CONCAT('Marca ', FLOOR(RAND() * 10)),
      ELT(FLOOR(RAND() * 3) + 1, 'Visita', 'Carga', 'Vendas')
    );
    SET i = i + 1;
  END WHILE;
END$$
DELIMITER ;

-- Chamar o procedimento armazenado para inserir os dados
CALL inserir_frota();
